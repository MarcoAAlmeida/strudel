/* `Drum_08 - waltz` */

// [waltz](https://youtu.be/5acGxguFiyo?t=131)
setcpm(60|3);

let waltz = s(`[hh ~ ~],[bd sd sd]`)
  .fast(1).bank(`rolandtr909`);


let wood = s(`bd*3`).sound(`east`);

arrange (
  [1, wood],
  [4, waltz],
).room(.1)
