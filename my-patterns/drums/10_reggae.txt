setcpm(60);

let reggae = s(`[~ hh cp hh], [~ ~ sd ~]`)
  .fast(1).bank(`rolandtr909`);


let wood = s(`bd*4`).sound(`east`);

arrange (
  [1, wood],
  [8, reggae],
).room(.1)
