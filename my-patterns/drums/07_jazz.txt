// [jazz](https://youtu.be/5acGxguFiyo?t=111)
setcpm(110/4);

let jazz = s(`[hh [hh ~ hh]],[~ ht]`)
  .fast(1).bank(`rolandtr909`);


let wood = s(`bd*4`).sound(`east`);

arrange (
  [1, wood],
  [4, jazz],
).room(.1)
